function wa(text: string) {
  return `https://wa.me/${PHONE}?text=${encodeURIComponent(text)}`;
}

function hires(src: string) {
  return src.startsWith('/images/') ? `/images/4k/${src.slice('/images/'.length)}.webp` : src;
}

function SmartImage({ asset, className = '', priority = false }: { asset: VisualAsset; className?: string; priority?: boolean }) {
  const fallback = asset.src;
  const fit = asset.fit ?? 'cover';
  return (
    <img
      src={hires(asset.src)}
      alt={asset.alt}
      loading={priority ? 'eager' : 'lazy'}
      decoding="async"
      fetchPriority={priority ? 'high' : 'auto'}
      sizes="(max-width: 768px) 100vw, 50vw"
      onError={(event) => {
        const image = event.currentTarget;
        if (image.dataset.fallback !== 'true') {
          image.dataset.fallback = 'true';
          image.src = fallback;
        }
      }}
      className={`${className} ${fit === 'contain' ? 'object-contain' : 'object-cover'}`}
      style={{ objectPosition: asset.position ?? 'center' }}
    />
  );
}

function productCount(route: PageRoute) {
  if (route === 'baustellenzubehoor') return constructionCategories.reduce((sum, category) => sum + category.products.length, 0);
  const key = categoryForRoute[route];
  return interiorCategories.find((category) => category.germanTitle === key)?.products.length ?? 0;
}
